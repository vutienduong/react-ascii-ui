Installation

npm install react-ascii-ui

Usage

import { AsciiSwitch } from 'react-ascii-ui';

export default function Example() {
  const [isOn, setIsOn] = useState(false);

  return (
    <AsciiSwitch 
      checked={isOn}
      onChange={(e) => setIsOn(e.target.checked)}
      label="Enable feature"
    />
  );
}
